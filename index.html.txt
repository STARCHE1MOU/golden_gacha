<script>
    const TIERS = {
        'N':    { name: 'Normal (Обычный)', color: '#888888' },
        'R':    { name: 'Rare (Редкий)', color: '#3498db' },
        'SR':   { name: 'Super Rare (Сверхредкий)', color: '#9b59b6' },
        'SSR':  { name: 'Super Super Rare (Топовый)', color: '#e67e22' },
        'SSRX': { name: 'Super Super Rare Extra (Высший)', color: '#ffd700' }
    };

    let FANDOMS_REGISTRY = []; 
    let CURRENT_LOTS = [];     

    window.addEventListener('DOMContentLoaded', () => {
        loadFandomsRegistry();
    });

    async function loadFandomsRegistry() {
        try {
            const response = await fetch('fandoms_list.txt');
            if (!response.ok) throw new Error('Не найден файл fandoms_list.txt');
            const text = await response.text();
            
            FANDOMS_REGISTRY = text.split('\n')
                .map(line => line.trim())
                .filter(line => line && !line.startsWith('#'))
                .map(line => {
                    const [id, name, weight] = line.split('|').map(s => s.trim());
                    return { id, name, weight: parseFloat(weight) || 1 };
                });

            const select = document.getElementById('fandomSelect');
            select.innerHTML = '<option value="" disabled selected>-- Выберите фандом для прокрутки --</option>';
            
            FANDOMS_REGISTRY.forEach(f => {
                const opt = document.createElement('option');
                opt.value = f.id;
                opt.innerText = f.name;
                select.appendChild(opt);
            });
            
            document.getElementById('statusInfo').innerText = `База данных синхронизирована. Доступно баннеров: ${FANDOMS_REGISTRY.length}`;
        } catch (err) {
            console.error(err);
            document.getElementById('statusInfo').innerText = "Ошибка: не удалось загрузить fandoms_list.txt";
        }
    }

    function rollBannerRoulette() {
        if (FANDOMS_REGISTRY.length === 0) return;
        
        const totalWeight = FANDOMS_REGISTRY.reduce((sum, f) => sum + f.weight, 0);
        let random = Math.random() * totalWeight;
        
        let selected = FANDOMS_REGISTRY[0];
        for (let f of FANDOMS_REGISTRY) {
            random -= f.weight;
            if (random <= 0) { selected = f; break; }
        }

        const resBox = document.getElementById('rouletteResult');
        resBox.style.color = 'var(--color-gold)';
        resBox.innerText = `🎯 Тебе выпал баннер: ${selected.name}`;
        
        const select = document.getElementById('fandomSelect');
        select.value = selected.id;
        onFandomChange();
    }

    async function onFandomChange() {
        const id = document.getElementById('fandomSelect').value;
        if (!id) return;

        try {
            const response = await fetch(`banners_data/${id}.txt`);
            if (!response.ok) throw new Error(`Не найден файл лотов banners_data/${id}.txt`);
            const text = await response.text();

            const lines = text.split('\n').map(l => l.trim());
            CURRENT_LOTS = [];
            let currentLot = null;

            for (let line of lines) {
                if (!line) continue;

                if (line.startsWith('#')) {
                    if (currentLot) {
                        currentLot.desc.push(line.substring(1).trim());
                    }
                } else {
                    const dotIndex = line.indexOf('.');
                    if (dotIndex !== -1) {
                        const tier = line.substring(0, dotIndex).trim().toUpperCase();
                        let rest = line.substring(dotIndex + 1).trim();
                        
                        let name = rest;
                        let inlineDesc = "";
                        
                        const hashIndex = rest.indexOf('#');
                        if (hashIndex !== -1) {
                            name = rest.substring(0, hashIndex).trim();
                            inlineDesc = rest.substring(hashIndex + 1).trim();
                        }

                        if (name.endsWith('.')) name = name.slice(0, -1).trim();

                        currentLot = { tier, name, desc: inlineDesc ? [inlineDesc] : [] };
                        CURRENT_LOTS.push(currentLot);
                    }
                }
            }
            document.getElementById('statusInfo').innerText = `Успешно загружено лотов из баннера: ${CURRENT_LOTS.length}`;
            document.getElementById('gachaResultsContainer').innerHTML = ''; 
        } catch (err) {
            console.error(err);
            alert('Не удалось загрузить лоты для этого баннера. Проверьте наличие файла в папке banners_data/.');
        }
    }

    function rollGacha(count) {
        if (CURRENT_LOTS.length === 0) {
            alert('Сначала выберите фандом из списка!');
            return;
        }

        const container = document.getElementById('gachaResultsContainer');
        container.innerHTML = '';

        let pool = [...CURRENT_LOTS];
        let rolls = Math.min(count, pool.length);

        for (let i = 0; i < rolls; i++) {
            const index = Math.floor(Math.random() * pool.length);
            const item = pool[index];
            
            pool.splice(index, 1);

            const card = document.createElement('div');
            card.className = 'gacha-card';
            
            const tierData = TIERS[item.tier] || { name: item.tier, color: '#ffffff' };
            card.style.borderColor = tierData.color;

            card.innerHTML = `
                <div class="card-tier" style="color: ${tierData.color};">[${tierData.name}]</div>
                <div class="card-name" style="color: ${tierData.color};">${item.name}</div>
                <div class="card-desc">${item.desc.join('\n') || 'Описание отсутствует.'}</div>
            `;
            container.appendChild(card);
        }
    }
